132 static uint16_t _bMBCRC16( uint8_t * pucFrame, uint16_t usLen )
133 {
134  uint8_t ucCRCHi = 0xFF;
135  uint8_t ucCRCLo = 0xFF;
136  int iIndex;
137 
138  while( usLen-- )
139  {
140  iIndex = ucCRCLo ^ *( pucFrame++ );
141  ucCRCLo = ( uint8_t )( ucCRCHi ^ aucCRCHi[iIndex] );
142  ucCRCHi = aucCRCLo[iIndex];
143  }
144  return ( uint16_t )( ucCRCHi << 8 | ucCRCLo );
145 }
